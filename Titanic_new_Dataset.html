    Age = cols[0]
    Pclass = cols[1]
    if pd.isnull(Age):
        if Pclass == 1:
            return a1     # or 38
        elif Pclass == 2:
            return a2    # or 29
        if Pclass == 3:
            return a3    # or 25
    else:
        return Age
